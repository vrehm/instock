Creating a new Product Image

Create a new product image

POST /admin/products/#{id}/images.xml

Create a new product image using a source URL that will be downloaded by Shopify

Request

           
<?xml version="1.0" encoding="UTF-8"?>
<image>
  <src>http://example.com/rails_logo.gif</src>
</image>

Response

           
HTTP/1.1 201 Created
<?xml version="1.0" encoding="UTF-8"?>
<product-image>
  <created-at type="datetime">2009-06-24T11:55:02-04:00</created-at>
  <id type="integer">1924445095</id>
  <position type="integer">2</position>
  <product-id type="integer">1706652214</product-id>
  <updated-at type="datetime">2009-06-24T11:55:02-04:00</updated-at>
  <src>http://static.shopify.com/s/files/1/6909/3384/products/rails_logo.gif?0</src>
</product-image>
