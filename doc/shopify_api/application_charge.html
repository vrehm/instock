Creating a new Application Charge

Create a new charge for a shop

POST /admin/application_charges.xml

Create a one-time charge called 'Super Duper Expensive action' for $100.00 USD

Request

           
<?xml version="1.0" encoding="UTF-8"?>
<application-charge>
  <name>Super Duper Expensive action</name>
  <price type="float">100.0</price>
  <return-url>http://google.com</return-url>
</application-charge>

Response

           
HTTP/1.1 201 Created
<?xml version="1.0" encoding="UTF-8"?>
<application-charge>
  <created-at type="datetime">2009-06-08T16:44:08-04:00</created-at>
  <id type="integer">1007331632</id>
  <name>Super Duper Expensive action</name>
  <price type="decimal">100.00</price>
  <status>pending</status>
  <updated-at type="datetime">2009-06-08T16:44:08-04:00</updated-at>
  <confirmation-url>http://apple.myshopify.com/admin/subscriptions/confirm_application_charge/1007331632?signature=BAhpBDCpCjw%3D--cc78d60ed0ba947c865a8a997709c837a924621f</confirmation-url>
</application-charge>
